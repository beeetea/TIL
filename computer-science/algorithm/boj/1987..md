```cpp
#include <iostream>
#include <algorithm>
#include <vector>
#include <cstdio>
using namespace std;
//int visited[20][20] = {0,};
int visited_apb[26] = {0,}; // 'A' == 65
int maxlen = -9999; // 최대 길이
int n_row, n_col; // row, col의 개수

void track(int r, int c, int depth, vector<vector<char>> &map)
{
    // 방문하지 않았던 정점이면?
    if (visited_apb[map[r][c]-65] == 0)
    {
        // 체크인 후
        visited_apb[map[r][c]-65] = 1;
        
        // depth 확장
        if (r!=0)
        {
            track(r-1, c, depth+1, map);
        }
        if (r!=n_row-1)
        {
            track(r+1, c, depth+1, map);
        }
        if (c!=0)
        {
            track(r, c-1, depth+1, map);
        }
        if (c!=n_col-1)
        {
            track(r, c+1, depth+1, map);
        }
        
        visited_apb[map[r][c]-65] = 0;
    }
    // 방문했던 정점이라면? maxlen 갱신
    else
    {
        if (maxlen<depth-1)
        {
            maxlen = depth-1;
            return;
        }
    }
    
    
}

int main()
{
    int r, c;
    cin >> r >> c;
    n_row = r;
    n_col = c;
    vector<vector<char>> map(r, vector<char>(c));
    for (int i=0; i<r; i++)
    {
        for (int j=0; j<c; j++)
        {
            scanf(" %c", &map[i][j]);
        }
        
        
    }
    
    track(0, 0, 1, map);
    
    cout << maxlen;
    
    return 0;
}
```
